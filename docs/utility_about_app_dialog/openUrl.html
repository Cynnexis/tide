Future<void> openUrl(String stringUrl) async {
  if (await url.canLaunch(stringUrl)) {
    await url.launch(stringUrl);
  } else if (kDebugMode) {
    dev.log("Couldn't launch URL: $stringUrl", name: "aboutApp.canLaunch");
  }
}
