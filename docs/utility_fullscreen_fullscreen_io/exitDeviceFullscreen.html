Future<void> exitDeviceFullscreen() async {
  if (_canUseFullscreen) {
    await SystemChrome.restoreSystemUIOverlays();
    await SystemChrome.setEnabledSystemUIMode(
      SystemUiMode.manual,
      overlays: SystemUiOverlay.values,
    );
  }
}
